## Data Model

### Prisma schema (`server/prisma/schema.prisma`)

```prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

model User {
  id           String   @id @default(uuid()) @db.Uuid
  name         String   @db.VarChar(120)
  email        String   @unique @db.VarChar(255)
  passwordHash String   @map("password_hash") @db.VarChar(255)
  createdAt    DateTime @default(now()) @map("created_at")
  updatedAt    DateTime @updatedAt @map("updated_at")

  sessions           Session[]
  passwordResetTokens PasswordResetToken[]

  @@map("users")
}

model Session {
  id        String   @id @default(uuid()) @db.Uuid
  userId    String   @map("user_id") @db.Uuid
  tokenHash String   @unique @map("token_hash") @db.VarChar(255)
  expiresAt DateTime @map("expires_at")
  createdAt DateTime @default(now()) @map("created_at")

  user User @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@map("sessions")
  @@index([userId])
  @@index([tokenHash])
}

model PasswordResetToken {
  id        String    @id @default(uuid()) @db.Uuid
  userId    String    @map("user_id") @db.Uuid
  tokenHash String    @unique @map("token_hash") @db.VarChar(255)
  expiresAt DateTime  @map("expires_at")
  usedAt    DateTime? @map("used_at")
  createdAt DateTime  @default(now()) @map("created_at")

  user User @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@map("password_reset_tokens")
  @@index([userId])
  @@index([tokenHash])
}

// ─── Future models (reserved for Phase 4+) ──────────────────────
// These models are defined here to ensure the initial migration
// produces a complete schema. Routes and service handlers ship later.
// See Existing UI Data Contracts in docs/plans/phase-3-authentication.md
// for the exact Zod schemas the frontend expects.

model Project {
  id          String   @id @default(uuid()) @db.Uuid
  name        String   @db.VarChar(200)
  description String   @default("") @db.VarChar(2000)
  status      String   @default("active") @db.VarChar(20)  // active | paused | completed | archived
  leadName    String   @map("lead_name") @db.VarChar(120)
  memberCount Int      @default(0) @map("member_count")
  createdAt   DateTime @default(now()) @map("created_at")
  updatedAt   DateTime @updatedAt @map("updated_at")

  @@map("projects")
  @@index([status])
}
```

> **Note:** `DashboardStats` and `RecentActivity` are not separate database tables. Dashboard stats are computed from aggregate queries (project counts, status distributions). Recent activity is event-sourced from project audit logs. The exact storage strategy is a Phase 4+ design decision.

### Design decisions

| Decision | Rationale |
|---|---|
| **Opaque tokens, not JWTs** | Tokens are random UUIDs stored in DB. Revocable on demand. No key rotation needed. No token reuse after logout. |
| **Store SHA-256 hash of token, not raw token** | Database compromise doesn't leak active sessions. Token is returned only once at creation time. |
| **`password_hash` uses bcrypt** | Cost factor 12. Industry standard. Slow enough to resist brute force, fast enough for auth flows. |
| **`onDelete: Cascade` on sessions/reset tokens** | Deleting a user removes all their sessions and reset tokens automatically. |
| **UUID primary keys** | No sequential ID enumeration. Safe for user-facing references. |

## API Endpoints

### Auth — `/api/auth/*`

All auth routes prefixed with `/api/auth`. Unauthenticated (except logout and session).

| Method | Path | Request Body | Response (200) | Errors | Description |
|---|---|---|---|---|---|
| POST | `/api/auth/signup` | `{ name, email, password }` | `{ user, session }` | 409 email exists, 400 validation | Create account + return session |
| POST | `/api/auth/login` | `{ email, password }` | `{ user, session }` | 401 invalid credentials | Authenticate + return session |
| POST | `/api/auth/logout` | — | `{ message }` | 401 | Revoke current session. Requires auth. |
| POST | `/api/auth/forgot-password` | `{ email }` | `{ message }` | — | Always 200. Creates reset token (if email exists). |
| POST | `/api/auth/reset-password` | `{ token, password }` | `{ user, session }` | 400 invalid/expired token | Reset password + return new session |
| GET | `/api/auth/session` | — | `{ user, session }` | 401 | Validate current session. Requires auth. |

#### Signup validation (Zod, server-side)

```typescript
export const SignupBodySchema = z.object({
  name: z.string().min(1, 'Name is required').max(120),
  email: z.string().email('Valid email is required'),
  password: z.string().min(8, 'Password must be at least 8 characters').max(200),
});
// No confirmPassword — server trusts the client's pre-validated match.
// If needed, server can verify: body.password === body.confirmPassword
```

#### Session response shape

```typescript
interface SessionResponse {
  user: {
    id: string;
    name: string;
    email: string;
  };
  session: {
    id: string;
    token: string;           // Opaque UUID, returned ONCE at creation
    expiresAt: string;       // ISO 8601
  };
}
```

### Users — `/api/users/*`

Requires authentication (session token in `Authorization` header).

| Method | Path | Request Body | Response (200) | Errors |
|---|---|---|---|---|
| GET | `/api/users/me` | — | `{ id, name, email, createdAt }` | 401 |
| PUT | `/api/users/me` | `{ name?, email? }` | `{ id, name, email }` | 401, 409 email exists |
| PUT | `/api/users/me/password` | `{ currentPassword, newPassword }` | `{ message }` | 401, 400 wrong current password |

### Error response shape

All errors return consistent JSON:

```typescript
interface ApiErrorResponse {
  error: {
    code: string;          // 'VALIDATION_ERROR' | 'UNAUTHORIZED' | 'CONFLICT' | 'NOT_FOUND' | 'INTERNAL_ERROR'
    message: string;
    details?: Record<string, string[]>;  // Field-level validation errors
  };
}
```

HTTP status codes: 400 (validation), 401 (unauthorized), 409 (conflict/duplicate), 404 (not found), 500 (internal).

## Express Server Architecture

### Middleware stack (in order)

```
1. cors()              — Allow Vite dev origin (http://localhost:5173)
2. express.json()      — Parse JSON bodies
3. Request logger      — Method, path, duration (morgan or custom)
4. Auth middleware     — Extract Bearer token, validate session, attach req.user
                        (applied selectively via route)
5. Validation middleware — Zod schema validation per route
6. Route handlers      — Controller functions
7. Error handler       — Catch-all, returns structured JSON error
```

### Auth middleware

```typescript
// src/middleware/auth.ts
// Extracts `Authorization: Bearer <token>` from header.
// Hashes token, looks up in sessions table.
// If valid + not expired → attaches user to request, calls next().
// If invalid/expired → returns 401 { error: { code: 'UNAUTHORIZED', message: '...' } }
//
// Routes that need optional auth (rare) can use a `optionalAuth` variant
// that attaches user if token valid but doesn't reject if missing.

export interface AuthenticatedRequest extends Express.Request {
  user: User;
  session: Session;
}
```

### Error handler

```typescript
// src/middleware/error.ts
// Catches all thrown/next(err) errors.
// Translates known error types to structured JSON responses:
//   - ZodError → 400 VALIDATION_ERROR with field details
//   - Prisma.PrismaClientKnownRequestError (unique constraint) → 409 CONFLICT
//   - ApiError → status code from error
//   - unknown → 500 INTERNAL_ERROR (logs full error, never exposes details to client)
```

### Route structure

```typescript
// src/routes/auth.ts
import { Router } from 'express';
import { z } from 'zod';
import { requireAuth } from '@/middleware/auth';
import { validate } from '@/middleware/validate';
import * as authService from '@/services/auth';

const router = Router();

// POST /api/auth/signup
router.post('/signup',
  validate(SignupBodySchema),
  async (req, res, next) => {
    try {
      const result = await authService.signup(req.body);
      res.status(201).json(result);
    } catch (err) {
      next(err);
    }
  },
);

// POST /api/auth/login
router.post('/login',
  validate(LoginBodySchema),
  async (req, res, next) => {
    try {
      const result = await authService.login(req.body);
      res.json(result);
    } catch (err) {
      next(err);
    }
  },
);

// POST /api/auth/logout
router.post('/logout', requireAuth, async (req, res, next) => {
  try {
    await authService.logout(req.session.id);
    res.json({ message: 'Logged out successfully.' });
  } catch (err) {
    next(err);
  }
});

// POST /api/auth/forgot-password
router.post('/forgot-password',
  validate(ForgotPasswordBodySchema),
  async (req, res, next) => {
    try {
      await authService.forgotPassword(req.body.email);
      // Always return 200 to prevent email enumeration
      res.json({ message: 'If an account exists, a reset link has been sent.' });
    } catch (err) {
      next(err);
    }
  },
);

// POST /api/auth/reset-password
router.post('/reset-password',
  validate(ResetPasswordBodySchema),
  async (req, res, next) => {
    try {
      const result = await authService.resetPassword(req.body);
      res.json(result);
    } catch (err) {
      next(err);
    }
  },
);

// GET /api/auth/session
router.get('/session', requireAuth, (req, res) => {
  const { user, session } = req as AuthenticatedRequest;
  res.json({
    user: { id: user.id, name: user.name, email: user.email },
    session: { id: session.id, expiresAt: session.expiresAt },
  });
});

export default router;
```

Note: `GET /api/auth/session` does NOT return the token — the token is stored client-side and sent back in the `Authorization` header. The session endpoint validates it and returns user + session metadata.

## Token Management

### Token lifecycle

```
Signup / Login
  │
  ├── Generate random UUID v4 (token)
  ├── Hash token with SHA-256 → store hash in sessions table
  ├── Return raw token to client (ONCE)
  │
  ▼
Client stores token in localStorage (tcs.auth)
  │
  ▼
Every API request: Authorization: Bearer <raw token>
  │
  ▼
Server: hash(raw token) → lookup in sessions table
  ├── Found + not expired → attach user, proceed
  ├── Found + expired → delete session, return 401
  └── Not found → return 401
  │
  ▼
Logout: delete session from DB
```

### Security considerations

| Concern | Mitigation |
|---|---|
| Token replay | Tokens are opaque UUIDs, single-use (no reuse after logout). HTTPS in production prevents capture. |
| Token theft from localStorage | XSS prevention (CSP headers, no `dangerouslySetInnerHTML`). HttpOnly cookie migration is a future ADR. |
| Brute force login | bcrypt cost factor 12. Rate limiting can be added later (express-rate-limit). |
| Timing attacks on login | Compare password hash with `crypto.timingSafeEqual` (bcrypt.compare does this internally). |
| Token expiration | 72-hour expiry. `GET /api/auth/session` returns expiry. Client can proactively refresh. |

### Test DB strategy

**Option A — Dedicated test database** (`tcsgon_test`)
- Run migrations once, truncate tables between test runs
- Fast, isolated, matches production schema
- Requires creating the test DB manually or in setup script

**Option B — SQLite in-memory**
- Prisma supports SQLite as datasource
- Switch `DATABASE_URL` to `file:./test.db` for tests
- Some Postgres features may differ (uuid type, etc.) — use `@default(uuid())` which works in both

**Recommended:** Option A (dedicated test Postgres DB) for accuracy. Add a `test:setup` script that runs migrations on the test DB.

## Risks

| Risk | Severity | Mitigation |
|---|---|---|
| Postgres not installed/running | High | Document first-time setup clearly with local PostgreSQL prerequisites. |
| Prisma migration conflicts between dev machines | Medium | Use `prisma migrate dev` (auto-detects drift). Document migration workflow. |
| bcrypt native module compilation fails on Windows | Medium | Use `bcryptjs` (pure JS, slower but no native deps) as fallback. |
| Express 5 API differences from Express 4 | Low | Pin `express@5.0.0` version. Document if any API differences affect middleware. |
| CORS errors in dev | Low | `corsOrigin` config already scoped to `http://localhost:5173`. |
| Token hash collision (SHA-256) | Negligible | SHA-256 collision probability is astronomically low (~2^-256). |

### `server/prisma/seed.ts`

```typescript
import { PrismaClient } from '@prisma/client';
import bcrypt from 'bcrypt';
import { v4 as uuid } from 'uuid';

const prisma = new PrismaClient();

async function main() {
  // Create admin user
  const passwordHash = await bcrypt.hash('password123', 12);
  
  const admin = await prisma.user.upsert({
    where: { email: 'admin@tcsgon.dev' },
    update: {},
    create: {
      id: uuid(),
      name: 'Admin User',
      email: 'admin@tcsgon.dev',
      passwordHash,
    },
  });

  // Create a permanent session for E2E tests (expires in 365 days)
  const token = uuid();
  const tokenHash = await hashToken(token);
  
  await prisma.session.create({
    data: {
      id: uuid(),
      userId: admin.id,
      tokenHash,
      expiresAt: new Date(Date.now() + 365 * 24 * 60 * 60 * 1000),
    },
  });

  console.log(`Admin user created: admin@tcsgon.dev / password123`);
  console.log(`E2E session token: ${token}`);
}

async function hashToken(token: string): Promise<string> {
  const encoder = new TextEncoder();
  const data = encoder.encode(token);
  const hashBuffer = await crypto.subtle.digest('SHA-256', data);
  const hashArray = Array.from(new Uint8Array(hashBuffer));
  return hashArray.map((b) => b.toString(16).padStart(2, '0')).join('');
}

main()
  .catch((e) => { console.error(e); process.exit(1); })
  .finally(() => prisma.$disconnect());
```
